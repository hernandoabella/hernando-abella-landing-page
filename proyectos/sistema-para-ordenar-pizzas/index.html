function CInputs(){
    
    let ch = [];
    let visible = document.getElementsByClassName("visible")[0];
    let inputs = visible.getElementsByTagName('input');

    for (let chkd=0; chkd < inputs.length; chkd++){
        if (inputs[chkd].checked) {
            ch.push(inputs[chkd]);
        }
    }

    return ch;
}

function check(){

    let ch = 0;
    let visible = document.getElementsByClassName("visible")[0];
    let inputs = visible.getElementsByTagName('input');

    for (let chkd=0; chkd < inputs.length; chkd++){
        if (inputs[chkd].checked) {
            ch += 1;
        }
    }

    return ch;
}

                for (let q=0; q < secCards.length; q++) {
